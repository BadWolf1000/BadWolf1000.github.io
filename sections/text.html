Text Processing

  • grep "text" file Search text $ grep "error" /var/log/syslog
    Jan 28 ERR: disk full
  • grep -r "text" dir/ Recursive search $ grep -r "password" /etc/
    /etc/shadow:root:$6$hash...
  • grep -i "text" file Case-insensitive search $ grep -i "error" log.txt
    ERROR: connection refused
    Error: timeout
  • grep -v "text" file Invert match $ grep -v "#" /etc/ssh/sshd_config
    Port 22
    PermitRootLogin no
  • grep -c "text" file Count matches $ grep -c "Failed" /var/log/auth.log
    47
  • grep -E "regex" file Extended regex $ grep -E "[0-9]{1,3}\.[0-9]{1,3}" access.log
    192.168.1.5 - GET /admin
  • sed 's/old/new/' file Replace text $ sed 's/http/https/' urls.txt
    https://example.com
  • sed 's/old/new/g' file Replace all occurrences $ sed 's/foo/bar/g' config.txt
    (all foo replaced with bar)
  • sed -i 's/old/new/g' file In-place replace $ sed -i 's/debug=true/debug=false/g' app.conf
    (file modified directly)
  • awk '{print $1}' file Process columns $ awk '{print $1}' access.log
    192.168.1.5
    10.0.0.3
  • awk -F: '{print $1}' file Custom delimiter $ awk -F: '{print $1}' /etc/passwd
    root
    kali
    www-data
  • sort file.txt Sort lines $ sort names.txt
    alice
    bob
    charlie
  • sort -u file.txt Sort and deduplicate $ sort -u ips.txt
    10.10.10.1
    10.10.10.5
    192.168.1.1
  • uniq file.txt Remove duplicates $ sort data.txt | uniq
    (adjacent duplicates removed)
  • cut -d: -f1 file Extract field $ cut -d: -f1 /etc/passwd
    root
    kali
    www-data
  • tr 'a-z' 'A-Z' Translate characters $ echo "hello" | tr 'a-z' 'A-Z'
    HELLO
