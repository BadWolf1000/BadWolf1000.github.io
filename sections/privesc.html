Privilege Escalation

  • sudo -l Check sudo rights $ sudo -l
    (root) NOPASSWD: /usr/bin/vim
  • find / -perm -4000 2>/dev/null Find SUID binaries $ find / -perm -4000 2>/dev/null
    /usr/bin/passwd
    /usr/bin/pkexec
  • find / -perm -2000 2>/dev/null Find SGID binaries $ find / -perm -2000 2>/dev/null
    /usr/bin/wall
    /usr/bin/crontab
  • getcap -r / 2>/dev/null List capabilities $ getcap -r / 2>/dev/null
    /usr/bin/python3 = cap_setuid+ep
  • cat /etc/crontab Check cron jobs $ cat /etc/crontab
    * * * * * root /opt/cleanup.sh
  • ls -la /etc/cron* List all cron dirs $ ls -la /etc/cron.d/
    -rw-r--r-- 1 root root backup_job
  • find / -writable -type f 2>/dev/null Find writable files $ find / -writable -type f 2>/dev/null
    /opt/backup.sh
    /tmp/notes.txt
  • cat /etc/passwd Check user accounts $ cat /etc/passwd | grep bash
    root:x:0:0::/root:/bin/bash
    admin:x:1001:1001::/home/admin:/bin/bash
  • env Check environment vars $ env
    PATH=/usr/local/bin:/usr/bin
    LD_PRELOAD=/tmp/evil.so
